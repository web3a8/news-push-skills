from typing import List
from sqlalchemy import desc

def get_articles(self, limit: int = 20, offset: int = 0,
                 source_id: Optional[int] = None) -> List[Article]:
    """
    获取文章列表，支持分页和筛选

    Args:
        limit: 每页数量
        offset: 偏移量
        source_id: 来源ID筛选（可选）

    Returns:
        文章列表
    """
    session = self.get_session()
    query = session.query(Article)

    if source_id:
        query = query.filter(Article.source_id == source_id)

    articles = query.order_by(desc(Article.fetched_at))\
                    .offset(offset)\
                    .limit(limit)\
                    .all()
    session.close()
    return articles
```

from sqlalchemy import func
from datetime import date

def get_statistics(self) -> dict:
    """
    获取统计信息

    Returns:
        包含 sources_count, today_articles, total_articles 的字典
    """
    session = self.get_session()

    stats = {
        "sources_count": session.query(Source).count(),
        "today_articles": session.query(Article)
            .filter(func.date(Article.fetched_at) == date.today())
            .count(),
        "total_articles": session.query(Article).count(),
    }
    session.close()
    return stats
```

        name = request.form.get("name", "").strip()
        url = request.form.get("url", "").strip()
        source_type = request.form.get("source_type", "rss")
        update_interval = request.form.get("update_interval", "60")

        # 验证
        if not name:
            flash("名称不能为空", "error")

        # URL 格式验证
        url_pattern = re.compile(
            r"^https?://"  # http:// or https://

from flask import Flask


def nl2br(text):
    """将换行符转换为 <br> 标签"""
    if text:
        return text.replace("\n", "<br>")
    return ""


def create_app():
    """
    Flask 应用工厂函数

    Returns:
        Flask 应用实例
    """
    app = Flask(__name__)

def update_source(self, source_id: int, **kwargs) -> bool:
    """
    更新新闻源

    Args:
        source_id: 新闻源 ID
        **kwargs: 要更新的字段

    Returns:
        是否更新成功
    """
    session = self.get_session()
    source = session.query(Source).filter(Source.id == source_id).first()

    if source:
        for key, value in kwargs.items():
            if hasattr(source, key):
                setattr(source, key, value)

        session.commit()
        session.close()
        return True

    session.close()
    return False


def delete_source(self, source_id: int) -> bool:
    """
    删除新闻源

    Args:
        source_id: 新闻源 ID

    Returns:
        是否删除成功
    """
    session = self.get_session()
    source = session.query(Source).filter(Source.id == source_id).first()

    if source:
        session.delete(source)
        session.commit()
        session.close()
        return True

    session.close()
    return False


def update_user(self, user_id: int, **kwargs) -> bool:
    """
    更新用户信息

    Args:
        user_id: 用户 ID
        **kwargs: 要更新的字段

    Returns:
        是否更新成功
    """
    session = self.get_session()
    user = session.query(User).filter(User.id == user_id).first()

    if user:
        for key, value in kwargs.items():
            if hasattr(user, key):
                # 如果是密码或 API 密钥，需要加密
                if key in ["smtp_password", "ai_api_key"] and value:
                    setattr(user, key, self.secure.encrypt(value))
                elif value:  # 如果值不为空则更新
                    setattr(user, key, value)

        session.commit()
        session.close()
        return True

    session.close()
    return False
```

"""系统设置路由"""

from flask import Blueprint, render_template, request, flash, redirect, url_for
from news_push.storage.database import DatabaseManager

settings_bp = Blueprint("settings", __name__)


@settings_bp.route("/settings", methods=["GET", "POST"])
def settings():
    """系统设置"""
    db = DatabaseManager()
    user = db.get_user(user_id=1)  # 默认单用户

    if not user:
        flash("用户不存在，请先运行初始化", "error")
        return redirect(url_for("home.index"))

    if request.method == "POST":
        # 获取表单数据
        email = request.form.get("email", "").strip()
        smtp_host = request.form.get("smtp_host", "").strip()
        smtp_port = request.form.get("smtp_port", "587")
        smtp_username = request.form.get("smtp_username", "").strip()
        smtp_password = request.form.get("smtp_password", "").strip()

        # 验证
        if not email:
            flash("邮箱不能为空", "error")
            return render_template("settings/form.html", user=user)

        if not smtp_host:
            flash("SMTP 主机不能为空", "error")
            return render_template("settings/form.html", user=user)

        try:
            smtp_port = int(smtp_port)
            if smtp_port < 1 or smtp_port > 65535:
                flash("SMTP 端口必须在 1-65535 之间", "error")
                return render_template("settings/form.html", user=user)
        except ValueError:
            flash("SMTP 端口必须是数字", "error")
            return render_template("settings/form.html", user=user)

        # 更新配置（密码加密存储）
        try:
            db.update_user(
                user_id=1,
                email=email,
                smtp_host=smtp_host,
                smtp_port=smtp_port,
                smtp_username=smtp_username,
                smtp_password=smtp_password if smtp_password else None
            )
            flash("设置保存成功", "success")
            user = db.get_user(user_id=1)  # 重新获取更新后的用户
        except Exception as e:
            flash(f"保存失败：{str(e)}", "error")

    return render_template("settings/form.html", user=user)
```
